Connect to MongoDB 
                    
                        import { MongoClient, Db } from "mongodb"
                        class MongoDBClient {
                            private uri = "mongodb://localhost:27017";
                            private dbName = "test";
                            public collectionName = "test";
                            public db: Db;
                            public async connect():Promise< Db> {
                                return new Promise< Db>((resolve, reject) => {
                                    MongoClient.connect(this.uri)
                                    .then((connection) => {
                                        this.db = connection.db(this.dbName);
                                        resolve(this.db);
                                    }).catch((error) => {
                                        reject(error)
                                    });
                                });
                            }
                        }
                        const client = new MongoDBClient();
                        export default client;

Create 
                    
                        import DBClient from "./database-client";

                        class App {
                        public async start() {
                            let db = await DBClient.connect();

                            let result =
                                await db.collection("products").insertOne({
                                topic: "camera", cost: 10
                            });

                            return new Promise< Object>((resolve, reject) => {
                                if(!result) {
                                    reject("Error creating object in mongo");
                                }
                                resolve(result);
                            });
                        }
                    }
                    
                    You can also use 'insertMany' method to insert several documents.

Create 
                    
                        import {format} from 'mysql';
                        import DBClient = require("./database-client");

                        class App {
                        public async start() {
                            try {
                                DBClient.connect()

                                sql = 'INSERT INTO posts SET ?';
                                post = { id: 1, title: 'Hello MySQL' };
                                sql = format(sql, post);

                                let query = DBClient.query(sql, function (err, result) {
                                    console.log("Success to insert: ",
                                                JSON.stringify(post));
                                });
                            } catch (error) {
                                console.log("Unable to connect to db");
                            }
                        }
                    }

Delete 
                    
                        import DBClient = require("./database-client");

                        class App {
                        public async start() {
                            try {
                                DBClient.connect()

                                let sql = 'DELETE FROM posts WHERE title = "wrong"';

                                DBClient.query(sql, function (err: mysql.QueryError, result: mysql.OkPacket) {
                                    console.log('deleted ' + result.affectedRows + ' rows');
                                });
                            } catch (error) {
                                console.log("Unable to connect to db");
                            }
                        }
                    }
