Connect to MongoDB 
                    
                        import { MongoClient, Db } from "mongodb"


                        class DBClient {
                            private url = "mongodb://localhost:27017/test";
                            public db: Db;

                            public async connect() {
                                this.db = await MongoClient.connect(this.url);
                                console.log("Connected to db");

                                return this.db;
                            }
                        }

                        export = new DBClient();

Create 
                    
                        import {format} from 'mysql';
                        import DBClient = require("./database-client");

                        class App {
                        public async start() {
                            try {
                                DBClient.connect()

                                sql = 'INSERT INTO posts SET ?';
                                post = { id: 1, title: 'Hello MySQL' };
                                sql = format(sql, post);

                                let query = DBClient.query(sql, function (err, result) {
                                    console.log("Success to insert: ",
                                                JSON.stringify(post));
                                });
                            } catch (error) {
                                console.log("Unable to connect to db");
                            }
                        }
                    }
Read 
                    
                        import DBClient = require("./database-client");

                        class App {
                        public async start() {
                            try {
                                DBClient.query('SELECT * from users', (err, rows, fields) => {

                            }

Delete 
                    
                        import DBClient = require("./database-client");

                        class App {
                        public async start() {
                            try {
                                DBClient.connect()

                                let sql = 'DELETE FROM posts WHERE title = "wrong"';

                                DBClient.query(sql, function (err: mysql.QueryError, result: mysql.OkPacket) {
                                    console.log('deleted ' + result.affectedRows + ' rows');
                                });
                            } catch (error) {
                                console.log("Unable to connect to db");
                            }
                        }
                    }
Events (result, error, fields, packet, end) 
                    
                        import DBClient = require("./database-client");

                        class App {
                        public async start() {
                            try {
                                DBClient.connect()

                                DBClient.on('error', function (err: MysqlError) {
                                    console.log('error raised : ' + err);
                                });
                            } catch (error) {
                                console.log("Unable to connect to db");
                            }
                        }
